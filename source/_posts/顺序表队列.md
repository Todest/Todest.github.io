# 顺序表队列
**以下为数据结构作业**
如有错误，请指出，谢谢！
<!--more-->
```cpp
#include<iostream>
#include<cstdio>
#include<cstdlib>
#define MAXSIZE 100
#define OK 1
#define ERROR 0
#define OVERFLOW 2
using namespace std;
typedef int Status;
typedef int ElemType;
struct node
{
    int num;
};
typedef struct
{
    node *elem;
    int length;
}SqList;
Status InitQueue(SqList &L)
{
    L.elem=new node[MAXSIZE];
    if(!L.elem)exit(OVERFLOW);
    L.length=0;
    return OK;
}
Status QueueShow(SqList &L)
{
    for(int i=0;i<L.length;i++)
        cout<<L.elem[i].num<<' ';
    cout<<endl;
    return OK;
}
Status EnQueue(SqList &L,node e)
{
    if(L.length==MAXSIZE)return ERROR;
    else
    {
        L.elem[L.length]=e;
        ++L.length;
        return OK;
    }
}
Status DeQueue(SqList &L)
{
    for(int i=1;i<L.length;i++)
        L.elem[i-1]=L.elem[i];
    L.length--;
    return 0;
}
void menu()
{
    cout<<"1.EnQueue"<<endl;
    cout<<"2.DeQueue"<<endl;
    cout<<"0.Exit"<<endl;
}
int main()
{
    SqList L;
    node e;
    int a,n;
    InitQueue(L);
    while(true)
    {
        system("CLS");
        menu();
        QueueShow(L);
        cin>>a;
        getchar();
        if(a==1)
        {
            cout<<"Please Input Number To Be Insert:";
            cin>>n;getchar();
            e.num=n;
            EnQueue(L,e);
        }
        else if(a==2)DeQueue(L);
        else if(a==0)break;
    }
    return 0;
}
```
